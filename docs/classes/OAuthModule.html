    Class OAuthModule

    OAuth Module for NestJS Provides OAuth authentication for multiple social providers

    Example

    // Synchronous registration
    OAuthModule.forRoot({
      providers: [
        {
          name: 'google',
          credentials: {
            client_id: process.env.GOOGLE_CLIENT_ID,
            client_secret: process.env.GOOGLE_CLIENT_SECRET,
            redirect_uri: process.env.GOOGLE_REDIRECT_URI
          }
        }
      ]
    })

    // Asynchronous registration
    OAuthModule.forRootAsync({
      useFactory: (configService: ConfigService) => ({
        providers: [
          {
            name: 'google',
            credentials: {
              client_id: configService.get('GOOGLE_CLIENT_ID'),
              client_secret: configService.get('GOOGLE_CLIENT_SECRET'),
              redirect_uri: configService.get('GOOGLE_REDIRECT_URI')
            }
          }
        ]
      }),
      inject: [ConfigService]
    })
